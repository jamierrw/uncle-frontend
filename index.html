  <head>

    <title>Uncle</title>

    <meta charset="UTF-8" />

  </head>

  <body style="font-family: sans-serif; text-align: center; padding-top: 50px;">

    <h1>Ask Uncle</h1>

    <form id="chatForm">

      <input type="text" id="question" placeholder="Ask a question..." style="width: 300px; padding: 10px;" />

      <button type="submit" style="padding: 10px;">Ask</button>

    </form>

    <p id="reply"></p>


    <script>

      const form = document.getElementById('chatForm');

      form.onsubmit = async (e) => {

        e.preventDefault();

        const question = document.getElementById('question').value;

        const response = await fetch("https://uncle-backend.onrender.com/ask", {

          method: "POST",

          headers: { "Content-Type": "application/json" },

          body: JSON.stringify({ question })

        });

        const data = await response.json();

        document.getElementById('reply').textContent = data.reply;

      };

    </script>

  </body>

</html>
